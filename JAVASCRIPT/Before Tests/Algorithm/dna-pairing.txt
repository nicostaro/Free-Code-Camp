function pairElement(str) {
  let one = str.split("");
  let two = [];
  
  for (let i =0; i<one.length;i++){
    two.push([one[i]]);
  }
  for (let j=0; j<two.length;j++){
    if (two[j][0]=="C"){
      two[j][1]="G";
    }
   else if (two[j][0]=="G"){
      two[j][1]="C";
  }
  else if (two[j][0]=="T"){
      two[j][1]="A";
  }
  else if (two[j][0]=="A"){
      two[j][1]="T";
  }
 }return two;
}
